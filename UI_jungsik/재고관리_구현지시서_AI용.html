2. 구현할 기능 명세 (확정)

핵심 요구사항:
설정(Settings)에서 재고관리 ON/OFF 토글을 추가한다.
• OFF (기본값): 현재 시스템 그대로, 변화 없음
• ON: 수확 재고가 있어야 납품/택배 등록 가능 (부족 시 경고, 강제차단 아님)

2.1 재고 계산 공식

현재 재고(crop_name별) =
  SUM(harvest_records.quantity WHERE farm_id = ?)     ← 수확 = 입고
  - SUM(sales_records.quantity WHERE farm_id = ?)     ← 납품+택배 = 출고
  - SUM(inventory_adjustments.quantity WHERE farm_id = ?) ← 폐기/보정

2.2 재고 경고 정책

상태조건UI등록 가능 여부
충분잔고 ≥ 입력량초록 배지✅ 허용
부족잔고 < 입력량빨강 경고✅ 허용 (경고만)
재고없음잔고 ≤ 0빨강 강경고✅ 허용 (강한 경고)

강제 차단하지 않는다. 현장에서 급한 상황에서도 등록이 가능해야 함.

ADD COLUMN IF NOT EXISTS inventory_warn_only BOOLEAN NOT NULL DEFAULT TRUE;

3.2 inventory_adjustments 테이블 생성

CREATE TABLE IF NOT EXISTS inventory_adjustments (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    farm_id UUID NOT NULL REFERENCES farms(id) ON DELETE CASCADE,
    crop_name TEXT NOT NULL,
    grade TEXT,
    adjustment_type TEXT NOT NULL
        CHECK (adjustment_type IN ('waste', 'correction', 'opening_stock')),
    quantity NUMERIC(10,2) NOT NULL,
    reason TEXT,
    adjusted_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    created_at TIMESTAMPTZ DEFAULT NOW()
);

-- RLS 활성화
ALTER TABLE inventory_adjustments ENABLE ROW LEVEL SECURITY;

-- RLS 정책: 농장 소유자만 접근
CREATE POLICY "farm_owner_inventory" ON inventory_adjustments
    FOR ALL
    USING (
        farm_id IN (
            SELECT id FROM farms WHERE owner_id = auth.uid()
        )
    );

-- 관리자 전체 접근 허용
CREATE POLICY "admin_inventory" ON inventory_adjustments
    FOR ALL
    USING (
        EXISTS (
            SELECT 1 FROM profiles WHERE id = auth.uid() AND role = 'admin'
        )
    );

4.2 hooks/useInventory.ts — 신규 파일 생성

"use client";
import { useState, useEffect } from "react";
import { supabase } from "@/lib/supabase";

// 특정 작물의 현재 재고 반환
export function useInventory(farmId: string | undefined, cropName: string | undefined) {
    const [stock, setStock] = useState<number | null>(null);
    const [loading, setLoading] = useState(false);

    useEffect(() => {
        if (!farmId || !cropName) return;
        fetchStock();
    }, [farmId, cropName]);

    const fetchStock = async () => {
        if (!farmId || !cropName) return;
        setLoading(true);

        const [harvestRes, salesRes, adjustRes] = await Promise.all([
            supabase.from('harvest_records')
                .select('quantity')
                .eq('farm_id', farmId)
                .eq('crop_name', cropName),
            supabase.from('sales_records')
                .select('quantity')
                .eq('farm_id', farmId)
                .eq('crop_name', cropName),
            supabase.from('inventory_adjustments')
                .select('quantity')
                .eq('farm_id', farmId)
                .eq('crop_name', cropName)
                .eq('adjustment_type', 'waste'),
        ]);

        const harvested = (harvestRes.data || []).reduce((s, r) => s + (r.quantity || 0), 0);
        const sold = (salesRes.data || []).reduce((s, r) => s + (r.quantity || 0), 0);
        const wasted = (adjustRes.data || []).reduce((s, r) => s + (r.quantity || 0), 0);

        setStock(Math.max(0, harvested - sold - wasted));
        setLoading(false);
    };

    return { stock, loading, refresh: fetchStock };
}

// 농장 전체 작물 재고 반환
export function useAllInventory(farmId: string | undefined) {
    const [stocks, setStocks] = useState<Record<string, number>>({});
    const [loading, setLoading] = useState(false);

    useEffect(() => {
        if (!farmId) return;
        fetchAllStocks();
    }, [farmId]);

    const fetchAllStocks = async () => {
        if (!farmId) return;
        setLoading(true);

        const [harvestRes, salesRes, adjustRes] = await Promise.all([
            supabase.from('harvest_records').select('crop_name, quantity').eq('farm_id', farmId),
            supabase.from('sales_records').select('crop_name, quantity').eq('farm_id', farmId),
            supabase.from('inventory_adjustments').select('crop_name, quantity').eq('farm_id', farmId).eq('adjustment_type', 'waste'),
        ]);

        const map: Record<string, number> = {};
        (harvestRes.data || []).forEach(r => { map[r.crop_name] = (map[r.crop_name] || 0) + r.quantity; });
        (salesRes.data || []).forEach(r => { map[r.crop_name] = (map[r.crop_name] || 0) - r.quantity; });
        (adjustRes.data || []).forEach(r => { map[r.crop_name] = (map[r.crop_name] || 0) - r.quantity; });

        Object.keys(map).forEach(k => { if (map[k] < 0) map[k] = 0; });
        setStocks(map);
        setLoading(false);
    };

    return { stocks, loading, refresh: fetchAllStocks };
}
